float ceil2(float A, float B)   
{
    preserve edx,ebx,eax;
    mov eax, A;
    mov ebx, B;
    mov edx, 10;
    fpwr ebx, edx;
    div eax, B;
    mov edx, eax;
    //edx = A/(10^^(B)); 
    FCEIL edx;
    mul edx, ebx;
    //`mov edx,edx*(10^^(B)); 
    return edx; 
    zap edx;
}

    fpwr ebx, edx;
    div eax, B;
    mov edx, eax;

    FINT edx;
    mul edx, ebx;
    //`mov edx,edx*(10^^(B));

    zap edx;
}

{

    fpwr ebx, edx;
